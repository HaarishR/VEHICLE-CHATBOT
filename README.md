# VEHICLE-CHATBOT


# 1. Navigate to your project folder
cd /path/to/your/project/folder

# 2. Initialize the folder as a Git repository
git init

# 3. Add all project files to the staging area
git add .

# 4. Commit the files with a descriptive message
git commit -m "Initial commit"

# 5. Rename the default branch to 'main'
git branch -M main

# 6. Link your local project to your GitHub repository
git remote add origin PASTE_YOUR_COPIED_URL_HERE

# 7. Push your code to GitHub
git push -u origin main
